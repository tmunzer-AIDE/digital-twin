```python
@dataclass(frozen=True)
class DhcpScope:
    """A SERVING dhcpd_config entry's range facts (GS25 lint surface).

    provider: "site" (switch-hosted site dhcpd_config) or a gateway device id
    (its OWN dhcpd_config). Identity is provider:network — exactly how
    dhcpd_config is keyed. vlan/subnet are resolution RESULTS (org namespace
    for gateway scopes, site networks for site scopes) and may be None when
    the namespace is blind; range fields are LITERAL config, None when absent
    or templated ({{var}}). Relay/none entries never become scopes (they are
    dhcp_sources material only — _dhcp_serves_scope, NOT _dhcp_active).
    """

    provider: str
    network: str
    vlan: int | None = None
    ip_start: str | None = None
    ip_end: str | None = None
    gateway: str | None = None
    subnet: str | None = None
    # True iff subnet INTENT exists but is unreadable (templated value) or
    # unknowable (unfetched org namespace). False when the namespace is
    # fetched and simply declares no subnet — no intent is NOT a blind spot
    # (a blanket note would PARTIAL-floor ordinary subnet-less networks).
    subnet_unresolved: bool = False
    meta: FactMeta = CONFIG_META

    @property
    def id(self) -> str:
        return f"{self.provider}:{self.network}"
```

```python
def _dhcp_trust(usage: Mapping[str, Any]) -> bool | None:
    """Tri-state DHCP-offer trust (GS25). The OAS marks allow_dhcpd itself
    tri-state: only the UNDEFINED value defers to the mode default. An empty
```

```python
def _snooping(eff: Mapping[str, Any]) -> tuple[str, ...] | None:
    """Device.dhcp_snooping: None = disabled; ("*",) = all_networks; else the
    enabled network names (site-network namespace)."""

    return tuple(sorted(str(n) for n in (cfg.get("networks") or ())))
```

```python
def _dhcp_serves_scope(entry: Any) -> bool:
    """Does this dhcpd entry OWN a scope (GS25)? Distinct from _dhcp_active
    ("is this a PATH"): relay-with-servers is an active path but owns no
    ip_start..ip_end — minting a range-less row would drag PARTIAL noise
    onto every normal relay config."""
    kind = str((entry or {}).get("type") or "local")
    return kind in _DHCP_SERVING_TYPES


def _literal_ip(value: Any) -> str | None:
    """Range/gateway fields are literal config; a templated {{var}} (or empty)
    is None — unknown, never parsed or guessed (doctrine f)."""
    if not value or "{{" in str(value):
        return None
    return str(value)
```

```python
    baseline_site_eff: Mapping[str, _Json], proposed_site_eff: Mapping[str, _Json]

    """A dhcpd range/gateway value the model cannot read ({{var}}) that the
    DELTA introduces or changes -> WARNING. Unlike bridge_priority (a GLOBAL
    election poisoned by either side), a templated range only poisons
    conclusions about that one scope — pre-existing unchanged templates stay
    silent here and degrade scope_lint coverage instead (GS25 spec)."""

    prop_cfg = proposed_site_eff.get("dhcpd_config") or {}

                    evidence={"network": str(name), "field": field, "value": str(value)},
```

```python
"""wired.dhcp.scope_lint — overlapping DHCP scope ranges + out-of-subnet scope
facts (GS25, MVP: CFG-DHCP-RNG / CFG-DHCP-CFG).

WARNING/REVIEW tier: external DHCP and placement unknowns mean ERROR would
over-claim (spec). Violation-specific parity (review r1/r3): pre-existing
demotion to INFO requires the SAME violation with the SAME values in baseline —
touching the hazard forfeits demotion. Unevaluable rows (templated/blind
fields) abstain with notes only when the delta touches scopes or a conclusion
was emitted (gateway_gap lesson: irrelevance never drags PARTIAL).
"""

from __future__ import annotations

import ipaddress
from itertools import combinations

from digital_twin.checks.base import CheckContext, CheckResult, Coverage, CoverageState, Status
from digital_twin.contracts import Finding, FindingCategory, FindingSource, Severity
from digital_twin.ir import (
    Capability,
    Confidence,
    ConfidenceLevel,
    DhcpScope,
    IRCapability,
    IRDiff,
    min_confidence,
)

_HIGH = Confidence(level=ConfidenceLevel.HIGH)


def _ip(value: str | None) -> int | None:
    if value is None:
        return None
    try:
        return int(ipaddress.ip_address(value))
    except ValueError:
        return None


def _range(s: DhcpScope) -> tuple[int, int] | None:
    lo, hi = _ip(s.ip_start), _ip(s.ip_end)
    if lo is None or hi is None:
        return None
    return (lo, hi) if lo <= hi else (hi, lo)


def _net(subnet: str | None) -> ipaddress.IPv4Network | ipaddress.IPv6Network | None:
    if subnet is None:
        return None
    try:
        return ipaddress.ip_network(subnet, strict=False)
    except ValueError:
        return None


def _overlaps(a: tuple[int, int], b: tuple[int, int]) -> bool:
    return a[0] <= b[1] and b[0] <= a[1]


def _subnet_violations(s: DhcpScope) -> tuple[str, ...]:
    net = _net(s.subnet)
    if net is None:
        return ()
    out = []
    for field in ("ip_start", "ip_end", "gateway"):
        v = _ip(getattr(s, field))
        if v is not None and ipaddress.ip_address(v) not in net:
            out.append(f"{field}={getattr(s, field)}")
    return tuple(out)


class DhcpScopeLintCheck:
    id = "wired.dhcp.scope_lint"
    title = "DHCP scope ranges collide or escape their subnet"
    domain = "wired.dhcp"
    default_severity = Severity.WARNING

    def requires(self) -> frozenset[Capability]:
        return frozenset({IRCapability.WIRED_L2})

    def applies_to(self, diff: IRDiff) -> bool:
        return any(diff.touches(k) for k in ("dhcp_scope", "vlan"))

    def run(self, ctx: CheckContext) -> CheckResult:
        base = {s.id: s for s in ctx.baseline.ir.dhcp_scopes}
        prop = {s.id: s for s in ctx.proposed.ir.dhcp_scopes}
        findings: list[Finding] = []

        # --- .overlap (pairwise, evaluable ranges only)
        ranged = {sid: r for sid, s in prop.items() if (r := _range(s)) is not None}
        for (ida, ra), (idb, rb) in combinations(sorted(ranged.items()), 2):
            if not _overlaps(ra, rb):
                continue
            ba, bb = base.get(ida), base.get(idb)
            preexisting = (
                ba is not None and bb is not None
                and (rba := _range(ba)) is not None and (rbb := _range(bb)) is not None
                and _overlaps(rba, rbb)
                and (ba.ip_start, ba.ip_end) == (prop[ida].ip_start, prop[ida].ip_end)
                and (bb.ip_start, bb.ip_end) == (prop[idb].ip_start, prop[idb].ip_end)
            )
            findings.append(self._finding(
                "overlap",
                Severity.INFO if preexisting else Severity.WARNING,
                f"DHCP scopes {ida} and {idb} hand out overlapping ranges"
                + (" (pre-existing, unchanged by the delta)" if preexisting else ""),
                {"scopes": [ida, idb]},
                (ida, idb),
            ))

        # --- .out_of_subnet
        for sid, s in sorted(prop.items()):
            bad = _subnet_violations(s)
            if not bad:
                continue
            b = base.get(sid)
            preexisting = (
                b is not None
                and _subnet_violations(b) == bad
                and b.subnet == s.subnet
            )
            findings.append(self._finding(
                "out_of_subnet",
                Severity.INFO if preexisting else Severity.WARNING,
                f"DHCP scope {sid}: {', '.join(bad)} outside the network subnet "
                f"{s.subnet}" + (" (pre-existing, unchanged)" if preexisting else ""),
                {"scope": sid, "violations": list(bad), "subnet": s.subnet},
                (sid,),
            ))

        # --- abstention: unevaluable rows, noted only when RELEVANT
        # (a changed scope cannot be cleared against a row we cannot read).
        # BOTH lint dimensions abstain visibly: unreadable ranges blind the
        # overlap lint; an unknown subnet blinds .out_of_subnet (spec).
        unevaluable = tuple(
            f"scope {sid}: range fields unreadable (templated or blind namespace) — "
            "overlap lint cannot clear changes against it"
            for sid, s in sorted(prop.items())
            if _range(s) is None
        ) + tuple(
            f"scope {sid}: owning network subnet is declared-but-unreadable or "
            "unknowable — subnet coherence is unevaluated for it"
            for sid, s in sorted(prop.items())
            if s.subnet_unresolved
        )
        conclusions = [f for f in findings if f.severity is not Severity.INFO]
        relevant = bool(conclusions) or ctx.diff.touches("dhcp_scope")
        notes = unevaluable if (unevaluable and relevant) else ()

        worst = Status.PASS
        if conclusions:
            worst = Status.WARN
        return CheckResult(
            check_id=self.id,
            status=worst,
            findings=tuple(findings),
            coverage=Coverage(
                state=CoverageState.PARTIAL if notes else CoverageState.COMPLETE,
                notes=notes,
            ),
            # gateway_gap precedent: confidence from CONCLUSIONS only — an
            # INFO-demoted pre-existing violation must never drag the result
            confidence=(
                min_confidence(*(f.confidence for f in conclusions))
                if conclusions else _HIGH
            ),
            reasoning="pairwise scope-range overlap + per-scope subnet coherence, "
                      "baseline-aware (violation-specific parity)",
        )

    def _finding(
        self, code: str, severity: Severity, message: str,
        evidence: dict, entities: tuple[str, ...],
    ) -> Finding:
        return Finding(
            source=FindingSource.CHECK,
            category=FindingCategory.NETWORK,
            code=f"{self.id}.{code}",
            severity=severity,
            confidence=_HIGH,
            message=message,
            affected_entities=entities,
            evidence=evidence,
        )
```

```python
"""wired.dhcp.snooping — snooping turned on with no trusted egress toward the
vlan's modeled DHCP source (GS25).

Path rule (spec): every modeled source must be reachable through at least one
egress whose FIRST HOP out of the snooping switch is a trusted port carrying
the vlan. ALL known egress candidates untrusted -> .untrusted_path WARNING.
Any candidate with UNKNOWN trust (or unknown carriage) on an otherwise
all-untrusted set -> abstain + PARTIAL (unknown is never untrusted). "site"
sources are unlocatable by design (per-device activation is unmodeled) ->
always PARTIAL for that vlan. Introduction is ACTIVITY (snooping enabled OR
the last trusted path lost), demotion requires the same blockage in baseline.
"""

from __future__ import annotations

import networkx as nx

from digital_twin.analysis.context import AnalysisContext
from digital_twin.checks.base import CheckContext, CheckResult, Coverage, CoverageState, Status
from digital_twin.contracts import Finding, FindingCategory, FindingSource, Severity
from digital_twin.ir import (
    IR,
    Capability,
    Confidence,
    ConfidenceLevel,
    IRCapability,
    IRDiff,
    min_confidence,
)
from digital_twin.ir.entities import DeviceRole, Port

_HIGH = Confidence(level=ConfidenceLevel.HIGH)
_BLIND_GATEWAY = Confidence(
    level=ConfidenceLevel.MEDIUM,
    reasons=("the source gateway's namespace is unmodeled — placement uncertain",),
)


def _snooped_vlans(ir: IR, did: str) -> tuple[int, ...]:
    dev = ir.devices.get(did)
    if dev is None or dev.dhcp_snooping is None:
        return ()
    if dev.dhcp_snooping == ("*",):
        return tuple(sorted(v for v, vl in ir.vlans.items() if vl.dhcp_sources))
    names = set(dev.dhcp_snooping)
    return tuple(sorted(v for v, vl in ir.vlans.items()
                        if vl.name in names and vl.dhcp_sources))


def _carries(port: Port, vlan: int) -> bool | None:
    """True/False = explicit carriage; None = vlan-blind (carriage unknown)."""
    if port.native_vlan is None and not port.tagged_vlans:
        return None
    return port.native_vlan == vlan or vlan in port.tagged_vlans


def _egress_trust(actx: AnalysisContext, ir: IR, did: str, vlan: int, source: str):
    """-> ("ok"|"blocked"|"unknown"|"unreachable", blocked_port_ids).

    Reachability beyond the first hop is VLAN-CONSTRAINED (review P1): a
    physical path that does not carry the vlan past the first hop cannot
    deliver offers — walking the bare L2 graph would false-SAFE through it.
    vlan_graph includes vlan-blind links as possible carriage (check its
    builder semantics); a source reachable only through blind links lands in
    the same honest place either way: never a trusted "ok" at HIGH.
    """
    g = actx.vlan_graph(vlan)
    if source not in g or did not in g:
        return "unreachable", ()
    pruned = nx.restricted_view(g, [did], [])
    statuses: list[tuple[bool | None, str]] = []
    for port in ir.ports.values():
        if port.device_id != did or port.disabled:
            continue
        links = [l for l in ir.links if port.id in (l.a_port, l.b_port)]
        for l in links:
            other_pid = l.b_port if l.a_port == port.id else l.a_port
            other_dev = ir.ports[other_pid].device_id
            reaches = other_dev == source or (
                other_dev in pruned and source in pruned
                and nx.has_path(pruned, other_dev, source)
            )
            if not reaches:
                continue
            carriage = _carries(port, vlan)
            if carriage is False:
                continue
            trust = None if carriage is None else port.dhcp_trusted
            statuses.append((trust, port.id))
    if not statuses:
        return "unreachable", ()
    if any(t is True for t, _ in statuses):
        return "ok", ()
    if any(t is None for t, _ in statuses):
        return "unknown", ()
    return "blocked", tuple(pid for _, pid in statuses)


class DhcpSnoopingCheck:
    id = "wired.dhcp.snooping"
    title = "DHCP snooping blocks the path to the vlan's DHCP source"
    domain = "wired.dhcp"
    default_severity = Severity.WARNING

    def requires(self) -> frozenset[Capability]:
        return frozenset({IRCapability.WIRED_L2})

    def applies_to(self, diff: IRDiff) -> bool:
        return any(diff.touches(k)
                   for k in ("device", "port", "dhcp_scope", "vlan", "link"))

    def run(self, ctx: CheckContext) -> CheckResult:
        base_ir, prop_ir = ctx.baseline.ir, ctx.proposed.ir
        findings: list[Finding] = []
        notes: list[str] = []
        for did, dev in sorted(prop_ir.devices.items()):
            if dev.role is not DeviceRole.SWITCH:
                continue
            for vlan in _snooped_vlans(prop_ir, did):
                sources = prop_ir.vlans[vlan].dhcp_sources
                has_site = "site" in sources
                if has_site:
                    notes.append(
                        f"vlan {vlan} on {did}: a site-hosted DHCP service is a "
                        "source but its device placement is unmodeled — snooping "
                        "impact on it cannot be evaluated"
                    )
                for source in sources:
                    if source == "site" or source == did:
                        continue  # unlocatable (noted) / local
                    state, blocked = _egress_trust(ctx.proposed, prop_ir, did, vlan, source)
                    if state == "unknown":
                        notes.append(
                            f"vlan {vlan} on {did}: an egress toward {source} has "
                            "unknown trust/carriage — cannot conclude offers drop"
                        )
                        continue
                    if state == "unreachable":
                        notes.append(
                            f"vlan {vlan} on {did}: DHCP source {source} is not "
                            "locatable in the modeled topology"
                        )
                        continue
                    if state == "ok":
                        continue
                    # blocked: demote only if the SAME blockage existed in baseline
                    pre = (
                        vlan in _snooped_vlans(base_ir, did)
                        and source in (base_ir.vlans.get(vlan).dhcp_sources
                                       if base_ir.vlans.get(vlan) else ())
                        and _egress_trust(ctx.baseline, base_ir, did, vlan, source)[0]
                        == "blocked"
                    )
                    src_dev = prop_ir.devices.get(source)
                    conf = (
                        _BLIND_GATEWAY
                        if src_dev is not None
                        and (src_dev.l3_unmodeled or src_dev.dhcp_unresolved)
                        else _HIGH
                    )
                    hedge = (
                        " — an unmodeled site-hosted service may still serve this vlan"
                        if has_site else ""
                    )
                    findings.append(Finding(
                        source=FindingSource.CHECK,
                        category=FindingCategory.NETWORK,
                        code=f"{self.id}.untrusted_path",
                        severity=Severity.INFO if pre else Severity.WARNING,
                        confidence=conf,
                        message=(
                            f"dhcp snooping on {did} vlan {vlan}: every egress "
                            f"toward DHCP source {source} is untrusted — offers "
                            f"are dropped at lease renewal{hedge}"
                            + (" (pre-existing, unchanged)" if pre else "")
                        ),
                        affected_entities=(did, str(vlan)),
                        evidence={"device": did, "vlan": vlan, "source": source,
                                  "untrusted_egress": list(blocked)},
                    ))
        conclusions = [f for f in findings if f.severity is not Severity.INFO]
        worst = Status.WARN if conclusions else Status.PASS
        return CheckResult(
            check_id=self.id,
            status=worst,
            findings=tuple(findings),
            coverage=Coverage(
                state=CoverageState.PARTIAL if notes else CoverageState.COMPLETE,
                notes=tuple(notes),
            ),
            # gateway_gap precedent (review P2): confidence from CONCLUSIONS
            # only — a pre-existing blockage demoted to INFO (possibly at the
            # blind-gateway MEDIUM cap) must never floor the result
            confidence=(
                min_confidence(*(f.confidence for f in conclusions))
                if conclusions else _HIGH
            ),
            reasoning="per snooping-switch/vlan: at least one trusted egress "
                      "toward every locatable modeled DHCP source",
        )
```
